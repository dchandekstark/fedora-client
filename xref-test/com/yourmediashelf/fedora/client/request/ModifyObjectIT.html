73      @Test
74      public void testModifyObjectState() throws Exception {
75          FedoraResponse response =
76                  modifyObject(testPid).state("I").logMessage(
77                          "testModifyObjectState()").execute();
78          assertEquals(200, response.getStatus());
79          String lastModifiedDate = response.getEntity(String.class);
80  
81          //validate the response
82          GetObjectProfileResponse getOP = getObjectProfile(testPid).execute();
83          assertEquals("I", getOP.getState());
84          assertEquals(lastModifiedDate, DateUtility.getXSDDateTime(getOP
85                  .getLastModifiedDate()));
86      }

88      @Test

90          DateTime lastModifiedDate =

92  
93          try {

95                      lastModifiedDate.minusHours(1)).logMessage(
96                      "testOptimisticLocking(): this request should fail")
97                      .execute();
98              fail("modifyObject succeeded, but should have failed with HTTP 409 Conflict");
99          } catch (FedoraClientException expected) {
100             assertEquals(409, expected.getStatus());
101         }
