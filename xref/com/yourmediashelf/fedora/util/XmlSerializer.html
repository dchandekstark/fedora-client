57   * <p>The various canonicalize methods use Apache XML Security to provide

94      /**
95       * Canonicalizes the input Document according to Canonical XML version 1.1.
96       *
97       * @param document
98       * @param out
99       * @throws Exception
100      */
101     public static void canonicalize(Document document, OutputStream out)
102             throws Exception {
103         Canonicalizer c14n =
104                 Canonicalizer
105                         .getInstance(Canonicalizer.ALGO_ID_C14N11_WITH_COMMENTS);
106         BufferedOutputStream bout = new BufferedOutputStream(out);
107         bout.write(c14n.canonicalizeSubtree(document));
108         bout.flush();
109     }

111     /**
112      * This isn't actually supported by any parser I've come across, so it's
113      * recommended to use the other canonicalize methods (which use Apache
114      * xmlsec)
115      *
116      * @param document
117      * @param out
118      */
119     public static void canonicalizeWithDOM3LS(Document document,
120             OutputStream out) {
121         DOMImplementationLS domImpl = getDOMImplementationLS(document);
122         LSSerializer lsSerializer = domImpl.createLSSerializer();
123         DOMConfiguration domConfiguration = lsSerializer.getDomConfig();
124         if (domConfiguration.canSetParameter("canonical-form", Boolean.TRUE)) {
125             lsSerializer.getDomConfig().setParameter("canonical-form",
126                     Boolean.TRUE);
127             LSOutput lsOutput = domImpl.createLSOutput();
128             lsOutput.setEncoding("UTF-8");
129             lsOutput.setByteStream(out);
130             lsSerializer.write(document, lsOutput);
131         } else {
132             throw new RuntimeException(
133                     "DOMConfiguration 'canonical-form' parameter isn't settable.");
134         }
135     }

137     public static void

139         DOMImplementationLS domImpl = getDOMImplementationLS(document);

141         DOMConfiguration domConfiguration = lsSerializer.getDomConfig();
142         if (domConfiguration.canSetParameter("format-pretty-print",

152                     "DOMConfiguration 'format-pretty-print' parameter isn't settable.");
153         }

156     private static DOMImplementationLS
157             getDOMImplementationLS(Document document) {
158         DOMImplementation domImplementation = document.getImplementation();
159         if (domImplementation.hasFeature("LS", "3.0") &&
160                 domImplementation.hasFeature("Core", "2.0")) {
161             DOMImplementationLS domImplementationLS;
162             try {
163                 domImplementationLS =
164                         (DOMImplementationLS) domImplementation.getFeature(
165                                 "LS", "3.0");
166             } catch (NoSuchMethodError nse) {
167                 logger.warn("Caught NoSuchMethodError for " +
168                         domImplementation.getClass().getName() +
169                         "#getFeature. " +
170                         "Trying fallback for DOMImplementationLS.");
171                 try {
172                     domImplementationLS = getDOMImplementationLS();
173                 } catch (Exception e) {
174                     throw new RuntimeException(e.getMessage(), e);
175                 }
176 
177             }
178             return domImplementationLS;
179         } else {
180             throw new RuntimeException(
181                     "DOM 3.0 LS and/or DOM 2.0 Core not supported.");
182         }
183     }

185     private static DOMImplementationLS getDOMImplementationLS()
186             throws ClassCastException, ClassNotFoundException,
187             InstantiationException, IllegalAccessException {
188         DOMImplementationRegistry registry;
189         registry = DOMImplementationRegistry.newInstance();
190         DOMImplementationLS impl =
191                 (DOMImplementationLS) registry.getDOMImplementation("LS");
192         return impl;
193     }

195     /**
196      * Get a new DOM Document object from parsing the specified InputStream.
197      *
198      * @param in
199      *        the InputStream to parse.
200      * @return a new DOM Document object.
201      * @throws ParserConfigurationException
202      *         if a DocumentBuilder cannot be created.
203      * @throws SAXException
204      *         If any parse errors occur.
205      * @throws IOException
206      *         If any IO errors occur.
207      */
208     private static Document getDocument(InputStream in)
209             throws ParserConfigurationException, SAXException, IOException {
210         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
211         factory.setNamespaceAware(true);
212         DocumentBuilder builder = factory.newDocumentBuilder();
213         // do not load external dtds
214         builder.setEntityResolver(new EntityResolver() {
215 
216             @Override
217             public InputSource resolveEntity(String publicId, String systemId)
218                     throws SAXException, IOException {
219                 return new InputSource(new StringReader(""));
220             }
221         });
222         return builder.parse(in);
223     }

225     protected static byte[] doc2bytes(Node node) {
226         try {
227             Source source = new DOMSource(node);
228             ByteArrayOutputStream out = new ByteArrayOutputStream();
229             Result result = new StreamResult(out);
230             TransformerFactory factory = TransformerFactory.newInstance();
231             Transformer transformer = factory.newTransformer();
232             transformer.transform(source, result);
233             return out.toByteArray();
234         } catch (TransformerConfigurationException e) {
235             e.printStackTrace();
236         } catch (TransformerException e) {
237             e.printStackTrace();
238         }
239         return null;
240     }

242     protected static String getStringFromDoc(Document document) {
243         DOMImplementationLS domImplementation =
244                 (DOMImplementationLS) document.getImplementation();
245         LSSerializer lsSerializer = domImplementation.createLSSerializer();
246         return lsSerializer.writeToString(document);
247     }

249     protected static Document string2document(String s) throws Exception {
250         return DocumentBuilderFactory.newInstance().newDocumentBuilder().parse(
251                 new InputSource(new StringReader(s)));
252     }
