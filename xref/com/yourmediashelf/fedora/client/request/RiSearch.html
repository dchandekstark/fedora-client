189     @Override
190     public RiSearchResponse execute(FedoraClient fedora)
191             throws FedoraClientException {
192         // set defaults
193         if (getFirstQueryParam("type") == null) {
194             type = "tuples";
195             addQueryParam("type", "tuples");
196         }
197         if (getFirstQueryParam("lang") == null) {
198             addQueryParam("lang", "sparql");
199         }
200         if (getFirstQueryParam("format") == null) {
201             if (type.equalsIgnoreCase("tuples")) {
202                 addQueryParam("format", "sparql");
203             } else if (type.equalsIgnoreCase("triples")) {
204                 addQueryParam("format", "n-triples");
205             }
206         }
207 
208         ClientResponse response = null;
209         String path = String.format("risearch");
210         WebResource wr = resource(fedora).path(path);
211 
212         // Check for a 302 (expected if baseUrl is http but Fedora is configured
213         // to require SSL
214         response = wr.head();
215         if (response.getStatus() == 302) {
216             URI newLocation = response.getLocation();
217             logger.warn("302 status for upload request: " + newLocation);
218             wr = resource(newLocation.toString());
219         }
220 
221         return new RiSearchResponse(wr.type(
222                 MediaType.APPLICATION_FORM_URLENCODED_TYPE).post(
223                 ClientResponse.class, getQueryParams()));
224     }

226     private void validateType(String type) {
227         if (type == null ||
228                 type.isEmpty() ||
229                 !(type.equalsIgnoreCase("tuples") || type
230                         .equalsIgnoreCase("triples"))) {
231             throw new IllegalArgumentException(
232                     "type must be one of tuples or triples");
233         }
234     }
235 
236     private void validateLang(String lang) {
237         if (lang == null || lang.isEmpty()) {
238             throw new IllegalArgumentException("lang cannot be null or empty");
239         } else if (type != null && type.equalsIgnoreCase("tuples")) {
240             if (!(lang.equalsIgnoreCase("sparql") || lang
241                     .equalsIgnoreCase("itql"))) {
242                 throw new IllegalArgumentException(
243                         "lang must be one of sparql "
244                                 + "or itql if type is tuples");
245             }
246         } else if (type != null && type.equalsIgnoreCase("triples")) {
247             if (!(lang.equalsIgnoreCase("sparql") ||
248                     lang.equalsIgnoreCase("itql") || lang
249                         .equalsIgnoreCase("spo"))) {
250                 throw new IllegalArgumentException(
251                         "lang must be one of sparql, "
252                                 + "itql, or spo if type is triples");
253             }
254         }
255     }
256 
257     private void validateFormat(String format) {
258         if (format == null || format.isEmpty()) {
259             throw new IllegalArgumentException("format cannot be null or empty");
260         } else if (type != null && type.equalsIgnoreCase("tuples")) {
261             if (!(format.equalsIgnoreCase("csv") ||
262                     format.equalsIgnoreCase("simple") ||
263                     format.equalsIgnoreCase("sparql") ||
264                     format.equalsIgnoreCase("tsv") || format
265                         .equalsIgnoreCase("count"))) {
266                 throw new IllegalArgumentException(
267                         "format must be one of "
268                                 + "csv, simple, sparql, tsv, or count if type is tuples");
269             }
270         } else if (type != null && type.equalsIgnoreCase("triples")) {
271             if (!(format.equalsIgnoreCase("n-triples") ||
272                     format.equalsIgnoreCase("Notation 3") ||
273                     format.equalsIgnoreCase("RDF/XML") ||
274                     format.equalsIgnoreCase("Turtle") || format
275                         .equalsIgnoreCase("count"))) {
276                 throw new IllegalArgumentException(
277                         "format must be one of "
278                                 + "n-triples, notation 3, rdf/xml, turtle, or count if "
279                                 + "type is triples");
280             }
281         }
282     }
283 
284     private void validateQuery(String query) {
285         if (query == null || query.isEmpty()) {
286             throw new IllegalArgumentException("query cannot be null or empty");
287         }
