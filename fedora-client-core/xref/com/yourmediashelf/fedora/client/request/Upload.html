20  package com.yourmediashelf.fedora.client.request;

22  import java.io.File;
23  import java.net.URI;
24  
25  import javax.ws.rs.core.MediaType;
26  
27  import com.sun.jersey.api.client.ClientResponse;
28  import com.sun.jersey.api.client.WebResource;
29  import com.sun.jersey.multipart.FormDataMultiPart;
30  import com.sun.jersey.multipart.MultiPart;
31  import com.sun.jersey.multipart.file.FileDataBodyPart;
32  import com.yourmediashelf.fedora.client.FedoraClient;
33  import com.yourmediashelf.fedora.client.FedoraClientException;
34  import com.yourmediashelf.fedora.client.response.UploadResponse;
35  
36  /**
37   * Builder for the Upload method.
38   *
39   * <p>The Fedora REST API upload method was introduced in Fedora 3.4</p>
40   *
41   * @author Edwin Shin
42   * @see "http://www.fedora-commons.org/jira/browse/FCREPO-687"
43   */
44  public class Upload extends FedoraRequest<Upload> {
45  
46      private final org.slf4j.Logger logger = org.slf4j.LoggerFactory
47              .getLogger(this.getClass());
48  
49      private final File file;
50  
51      /**
52       * @param file
53       *        the file to upload
54       */
55      public Upload(File file) {
56          this.file = file;
57      }
58  
59      @Override
60      public UploadResponse execute() throws FedoraClientException {
61          return (UploadResponse) super.execute();
62      }
63  
64      @Override
65      public UploadResponse execute(FedoraClient fedora)
66              throws FedoraClientException {
67          ClientResponse response = null;
68          String path = String.format("upload");
69          WebResource wr = resource(fedora).path(path);
70  
71          MediaType mediaType = MediaType.valueOf(fedora.getMimeType(file));
72          MultiPart multiPart =
73                  new FormDataMultiPart().bodyPart(new FileDataBodyPart("file",
74                          file, mediaType));
75  
76          // Check for a 302 (expected if baseUrl is http but Fedora is configured
77          // to require SSL
78          response = wr.head();
79          if (response.getStatus() == 302) {
80              URI newLocation = response.getLocation();
81              logger.warn("302 status for upload request: " + newLocation);
82              wr = resource(newLocation.toString());
83          }
84  
85          response =
86                  wr.queryParams(getQueryParams()).type(
87                          MediaType.MULTIPART_FORM_DATA_TYPE).post(
88                          ClientResponse.class, multiPart);
89          return new UploadResponse(response);
90      }
91  }
