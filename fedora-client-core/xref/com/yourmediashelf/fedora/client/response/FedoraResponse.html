20  package com.yourmediashelf.fedora.client.response;

22  import java.io.InputStream;
23  
24  import com.yourmediashelf.fedora.client.request.FedoraRequest;

26  /**
27   * <p>A wrapper for the HTTP response to a {@link FedoraRequest}.</p>
28   *
29   * <p>Request-specific FedoraResponse implementations should be generally
30   * preferred as they provide convenience methods specific to the kind of
31   * response. For example, {@link GetNextPIDResponse} provides a convenience
32   * method that parses the HTTP response an returns a List of the requested
33   * pids.</p>
34   *
35   * <p>Note that {@link #getEntity(Class)} and {@link #getEntityInputStream()}
36   * provide access to the underlying response entity, if desired. However, using
37   * these methods and the convenience methods that parse the response entity are
38   * mutually exclusive, as either will close the underlying InputStream once
39   * accessed.</p>
40   *
41   * <p>Implementations of FedoraResponse MUST throw a FedoraClientException if
42   * the HTTP status code of the response is >= 400.</p>
43   *
44   * @author Edwin Shin
45   */
46  public interface FedoraResponse {
47  
48      /**
49       *
50       * @return the HTTP Status Code of the response.
51       */
52      public int getStatus();
53  
54      /**
55       * <p>Gets the raw response entity.</p>
56       *
57       * @return the entity response
58       */
59      public InputStream getEntityInputStream();
60  
61      /**
62       * <p>Gets the raw response entity.</p>
63       *
64       * @return the entity response
65       */
66      public <T> T getEntity(Class<T> c);
67  
68      /**
69       * <p>Get the media type of the response;
70       * @return the media type.
71       */
72      public String getType();
73  
74      /**
75       * <p>Close the response.
76       * <p>The entity input stream is closed;
77       */
78      public void close();
79  
80  }
