97      /**
98       * Helper method that returns an XHTML document, allowing xmlunit assertions
99       * to be made against badly formed HTML source.
100      *
101      * @param html HTML that may not be well-formed
102      * @return a DOM document that has converted the 'plain' HTML source into valid 'XHTML' 
103      * @throws Exception
104      */
105     private Document buildTestDocument(String html) throws Exception {
106         TolerantSaxDocumentBuilder tolerantSaxDocumentBuilder =
107                 new TolerantSaxDocumentBuilder(XMLUnit.newTestParser());
108         HTMLDocumentBuilder htmlDocumentBuilder =
109                 new HTMLDocumentBuilder(tolerantSaxDocumentBuilder);
110         return htmlDocumentBuilder.parse(html);
111     }
