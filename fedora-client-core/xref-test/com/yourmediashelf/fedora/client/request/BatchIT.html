150     protected void getBatchDatastreams(Set<String> pids, Set<String> dsids) throws Exception {
151         BatchResponse response =
152                 new BatchGetDatastreams().pids(pids).dsids(dsids).execute();
153         assertEquals(200, response.getStatus());

179     /**
180      * Build a List of identifiers that match the provided regex.
181      *
182      */
183     protected static Set<String> matchIds(Collection<String> ids, String regex, int limit) {
184         Set<String> matchedIds = new HashSet<String>();
185         for (String id : ids) {
186             if (id.matches(regex)) {
187                 matchedIds.add(id);
188                 if (matchedIds.size() == limit) {
189                     break;
190                 }
191             }
192         }
193         return matchedIds;
194     }

204     /**
205      * Create a String of the requested size (in kilobytes).
206      * 
207      * @param msgSize size in kilobytes
208      * @return a String of 'a's
209      */
210     private static String createDataSize(int msgSize) {
211         // Java chars are 2 bytes
212         msgSize = msgSize / 2;
213         msgSize = msgSize * 1024;
214         StringBuilder sb = new StringBuilder(msgSize);
215         for (int i = 0; i < msgSize; i++) {
216             sb.append('a');
217         }
218         return sb.toString();
219     }
220     
221     private static String listAsCSV(Collection<String> list) {
222         String separator = ",";
223         StringBuilder sb = new StringBuilder();
224         for (String s : list) {
225             sb.append(s);
226             sb.append(separator);
227         }
228         return sb.toString().substring(0, sb.length()-separator.length());
229     }
